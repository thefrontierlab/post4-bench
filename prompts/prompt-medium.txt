def estimate_tokens(text: str, tokenizer_name: str = "gpt2") -> int:
    """Approximate token count for a string without loading a real tokenizer.

    Uses a simple heuristic based on whitespace splitting and average
    bytes-per-token observed across common English corpora.
    """
    import re

    if not text:
        return 0

    # Split on whitespace and common punctuation
    parts = re.split(r"[\s\.,;:!?\(\)\[\]\{\}\"\']+", text)
    parts = [p for p in parts if p]

    # Heuristic: words longer than 4 chars often get split into 2 tokens
    # by BPE; shorter words map to one token.
    estimate = 0
    for p in parts:
        if len(p) <= 4:
            estimate += 1
        else:
            estimate += len(p) // 4 + 1

    # Add overhead for special tokens (BOS, EOS, etc.)
    overhead = {"gpt2": 2, "llama": 1, "qwen": 3}.get(tokenizer_name, 2)
    return estimate + overhead
